### Install Tailwind CSS

<CodeCopy 
language="markdown"
code={`npm install -D tailwindcss postcss autoprefixer
npx tailwindcss init -p`}
/>

### Configure Tailwind

Add the following content in your Tailwind configuration File.

<CodeCopy 
language="typescript"
code={
  `import type { Config } from "tailwindcss"
const config = {
  darkMode: ["class"],
  content: [
    './pages/**/*.{js,ts,jsx,tsx,mdx}',
    './components/**/*.{ts,tsx}',
    './app/**/*.{ts,tsx}',
    './src/**/*.{ts,tsx}',
    './lib/**/*.{js,ts,jsx,tsx}'
  ],
  prefix: "",
  theme: {
    extend: {
      backgroundColor: {
        "ghost-bg": "ghost-bg",
        "ghost-text": "ghost-text",
        "ghost-hover": "ghost-hover",
      },
    },
  },
} satisfies Config
export default config`}

fileName = "tailwind.config.ts"
/>

### Add the Tailwind directives to your CSS

<CodeCopy 
language="css"
    code={`@tailwind base;
@tailwind components;
@tailwind utilities;`}

fileName="globals.css"

/>
